## Usage

```tsx
import { Autocomplete } from "<path-to-ui-components>/auto-complete";
```

```tsx
<Autocomplete
  options={frameworks}
  placeholder="Select framework..."
  emptyMessage="No framework found."
  value={selected}
  onValueChange={(value) => setSelected(value)}
/>
```
