## Usage

```tsx

  InfiniteScroll,
  InfiniteScrollList,
} from "<path-to-ui-components>/infinite-scroll";
```

```tsx
return (
    <InfiniteScroll
      count={count}
      estimatedHeight={100}
      overScan={10}
      itemSpacing={itemSpacing}
    >
       <InfiniteScrollList>
        {(index) => {
          return <div>{sentences[index]}</div>;
```
