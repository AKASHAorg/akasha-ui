## Usage

```tsx
import { DuplexButton } from "<path-to-ui-components>/duplex-button" ;
```

```tsx
<DuplexButton active={isSubscribed}>
  <DuplexButtonActive onClick={() => setIsSubscribed(false)}>
    <DuplexButtonActiveDefault>Subscribed</DuplexButtonActiveDefault>
    <DuplexButtonActiveHover>Unsubscribe</DuplexButtonActiveHover>
  </DuplexButtonActive>
  <DuplexButtonInactive onClick={() => setIsSubscribed(true)}>
    Subscribe
  </DuplexButtonInactive>
</DuplexButton>
```
