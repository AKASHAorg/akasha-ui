## Usage

```tsx
import {
  ProfileAvatar,
  ProfileAvatarFallback,
  ProfileAvatarImage,
} from "<path-to-ui-components>/profile-avatar";
```

```tsx
<ProfileAvatar>
  <ProfileAvatarImage src="https://github.com/shadcn.png" />
  <ProfileAvatarFallback>CN</ProfileAvatarFallback>
</ProfileAvatar>
```
